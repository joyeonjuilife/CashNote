use cashnote_db;
CREATE TABLE income (
    id INT AUTO_INCREMENT PRIMARY KEY,
    amount INT,
    category VARCHAR(50),
    memo VARCHAR(255),
    date DATE
);

CREATE TABLE expense (
    id INT AUTO_INCREMENT PRIMARY KEY,
    amount INT,
    category VARCHAR(50),
    memo VARCHAR(255),
    date DATE
);
create table category(
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(50) NOT NULL,
  type ENUM('income', 'expense') NOT NULL
);
